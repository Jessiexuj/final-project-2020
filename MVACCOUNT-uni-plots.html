Fdata = read.csv('FinalData - Sheet1.csv', header = TRUE)
length(unique(Fdata$MVACCOUNT))
## [1] 181
summary(Fdata$MVACCOUNT)
##      Min.   1st Qu.    Median      Mean   3rd Qu.      Max. 
## -2.202045 -0.843508  0.060541 -0.006007  0.803288  1.732547
library(ggplot2)
base= ggplot(Fdata,aes(x = MVACCOUNT))  
h1= base + geom_histogram(binwidth = 0.05)

h1

d1= base + stat_density(geom = "line")
d1

theMean=mean(Fdata$MVACCOUNT, na.rm = T)
theMedian=median(Fdata$MVACCOUNT, na.rm = T)

d1 + geom_vline(xintercept = theMedian,color='blue') +
     geom_vline(xintercept =theMean,color='red') 

d1 + geom_vline(aes(xintercept = theMedian,color='median')) +
     geom_vline(aes(xintercept =theMean,color='mean')) +
     scale_color_manual(name = "Central Measures",
                        values = c(median = "blue", mean = "red"),
                        breaks = c('median','mean'),
                       labels=c("MEDIAN", "MEAN"))

base= ggplot(Fdata,aes(y = MVACCOUNT))  
b1= base + geom_boxplot()

b1 + coord_flip()
